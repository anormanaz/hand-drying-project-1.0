Parameters

General parameters

#pulmonary ventilation rate (minutes))
IR <- rtrunc(iterations,"norm", a = 0, b = Inf, mean = 0.012, sd = 0.0025)

#time in minutes 
t_occupancy <- 2.2933

#viral air concentrations
c_air_pt_low <- 115.7407
c_air_recessed_rw <- 370.3704
c_air_surfmount_rw <- 105.8201
c_air_nonhepa_rw <- 238.9486

H1N1_alpha <- 9.04E-01
H1N1_n50 <- 1.25E+06

H3N2_alpha <- 4.29E-01
H3N2_n50 <- 6.66E+05

Functions

dose_function <- function(C){
  C * IR * t_occupancy
}

beta_Poisson_function <- function(dose, alpha, n50 ){
  1 - ((1 + dose * (2^(1/alpha) - 1) / n50)^(-alpha))
}

adeno_k <- 6.07E-01
exponential_function <- function(k,dose){
  1 - exp(-k * dose )
}
get_summary_stats <- function(x) {
 data.frame(
  Min = sprintf("%.2E", min(x)),
  Median = sprintf("%.2E", median(x)),
  Max = sprintf("%.2E", max(x)),
  Mean = sprintf("%.2E", mean(x))
)
}
